void Unweighted(Table t) {
  int currDist;
  Vertex v,w;
  for (currDist=0;currDist<Num; currDist++) {
    for each vertex v {
      if (!t[v].Known && t[v].Disk == currDist) {
        t[v].known = 1;
        for each w adjacent to v {
          if (t[w].dist == Infinity) {
            t[w].dist = currDist+1;
            t[w].path = v;
          }
        }
      }
    }
  }
}
